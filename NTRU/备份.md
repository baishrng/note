    /**
     * 通过 N 对文件进行分组
     * @author 白生
     * @date 2024/1/26 19:36
     */
    public static ArrayList<int[]> groupByN(int N, String filename) throws IOException {
        int groupSizeBit = 1;  // 分组大小(bit)
        ArrayList<int[]> dataList = new ArrayList<int[]>();

        /* 打开文件 */
        File file = new File(filename);
        FileInputStream fin = new FileInputStream(file);
        BufferedInputStream bufferedInputStream = new BufferedInputStream(fin);

        /* 计算分组大小 */
        while( groupSizeBit < N )
            groupSizeBit = groupSizeBit * 2;
        groupSizeBit = groupSizeBit / 2;

//        System.out.println(groupSizeBit);

        int groupSizeByte = groupSizeBit / 8;   // 分组大小(byte)

        byte[] buffer = new byte[groupSizeByte];    // 缓冲区

        int dataLength = (int) file.length();       // 数据总字节数

//        System.out.println(groupSizeByte);

        String dataString = "";

        int[] dataInt = new int[N+1];

        int readLength = 0;     // 读取字节长度

        /* 读取数据 */
        while( dataLength > 0 ){
            if( dataLength > groupSizeByte )
                readLength = groupSizeByte;
            else
                readLength = dataLength;

            dataString = "";
            bufferedInputStream.read(buffer, 0, readLength);

            /* 填充 */
            if( readLength != groupSizeByte ){
                int paddingBytes = groupSizeByte - readLength;
                for(int i=readLength; i<groupSizeByte; i++)
                    buffer[i] = (byte) paddingBytes;
            }

            // 字节转二进制字符串
            for(int i=0; i<groupSizeByte; i++){
                dataString = dataString + Tools.byteToBinary(buffer[i]);
            }

            // 字符串转int型数组
            char[] dataChars = dataString.toCharArray();
            for(int i=0; i<dataChars.length; i++)
                dataInt[i] = Character.getNumericValue(dataChars[i]);


            dataList.add(dataInt.clone());

            dataLength = dataLength - groupSizeByte;
        }

        bufferedInputStream.close();
        fin.close();

        return dataList;
    }

    /**
     * 文件还原
     * @author 白生
     * @date 2024/2/2 13:32
     */
    public static int reduction(int N, ArrayList<int[]> dataList, String filename) throws IOException {
        File file = new File(filename);
        FileOutputStream fout = new FileOutputStream(file);
        BufferedOutputStream bufferedOutputStream = new BufferedOutputStream(fout);

        /* 计算分组大小 */
        int groupSizeBit = 1;
        while( groupSizeBit < N )
            groupSizeBit = groupSizeBit * 2;
        groupSizeBit = groupSizeBit / 2;
        int groupSizeByte = groupSizeBit / 8;   // 分组大小(byte)

        String dataString = "";
        byte[] buffer = new byte[groupSizeByte];

        int writeLength = groupSizeByte;    // 写入字节长度

        for(int i=0; i<dataList.size(); i++){
            /* int数组 转 字符串 */
            dataString = Arrays.toString(dataList.get(i));
            dataString = dataString.substring(1, dataString.length()-1);
            dataString = dataString.replace(", ", "");
            dataString = dataString.substring(0,groupSizeBit);

            /* 二进制字符串转字节 */
            for(int j=0; j<groupSizeByte; j++){
                buffer[j] = Tools.binaryToByte(dataString.substring(j*8, j*8+8));
            }

            /* 去掉填充部分 */
            if( i == dataList.size()-1 ){
                int paddingBytes = buffer[groupSizeByte-1];
                if( paddingBytes != 1 ){
                    if( buffer[groupSizeByte-1] == buffer[groupSizeByte-2] )
                        writeLength = writeLength - paddingBytes;
                }
                else
                    writeLength = writeLength - paddingBytes;
            }

            bufferedOutputStream.write(buffer, 0, writeLength);     // 向文件写入数据
        }

        bufferedOutputStream.close();
        fout.close();

        return 0;
    }

                else

class DecryptThread implements Runnable{
//    private RandomAccessFile out;
    private int N;
    private int p;
    private int q;
    private int d;
    private int[] fx;
    private int[] gx;
    private ArrayList<int[]> cList;
    private String filename;
    private int groupId;
    private int state;

    public DecryptThread(int n, int p, int q, int d, int[] fx, int[] gx, ArrayList<int[]> cList, String filename, int groupId, int state) {
        N = n;
        this.p = p;
        this.q = q;
        this.d = d;
        this.fx = fx;
        this.gx = gx;
        this.cList = cList;
        this.filename = filename;
        this.groupId = groupId;
        this.state = state;
    }

    @Override
    public void run() {
        ArrayList<int[]> mList = new ArrayList<>();
        int[] mx = new int[N+1];
        Lock lock = new ReentrantLock();

        for(int[] as : cList){
            Tools.initPolynomial(mx);
            Tools.decryptPlaintext(N, p ,q, d, fx, gx, as, mx);
            mList.add(mx.clone());
        }

        try {
            write(N, filename, mList, groupId, state);
        } catch (IOException e) {
            throw new RuntimeException(e);
        }

        System.out.println("子线程"+groupId+"结束");
    }


    public static int write(int N, String filename, ArrayList<int[]> dataList, int groupId, int state) throws IOException {
        RandomAccessFile out = new RandomAccessFile(filename, "rw");

        /* 计算分组大小 */
        int groupSizeBit = 1;
        while( groupSizeBit < N )
            groupSizeBit = groupSizeBit * 2;
        groupSizeBit = groupSizeBit / 2;
        int groupSizeByte = groupSizeBit / 8;   // 分组大小(byte)

        String dataString = "";
        byte[] buffer = new byte[groupSizeByte];

        int writeLength = groupSizeByte;    // 写入字节长度

        /* 移动文件指针 */
        long seekBytes = (long) groupSizeByte * 1024 * groupId;
        out.seek(seekBytes);

        /* 转换并写入 */
        for(int i=0; i<dataList.size(); i++){
            /* int数组 转 字符串 */
            dataString = Arrays.toString(dataList.get(i));
            dataString = dataString.substring(1, dataString.length()-1);
            dataString = dataString.replace(", ", "");
            dataString = dataString.substring(0,groupSizeBit);

            /* 二进制字符串转字节 */
            for(int j=0; j<groupSizeByte; j++){
                buffer[j] = Tools.binaryToByte(dataString.substring(j*8, j*8+8));
            }

            /* 去掉填充部分 */
            if( state == 1 && i == dataList.size()-1 ){
                int paddingBytes = buffer[groupSizeByte-1];
                if( paddingBytes != 1 ){
                    if( buffer[groupSizeByte-1] == buffer[groupSizeByte-2] )
                        writeLength = writeLength - paddingBytes;
                }
                else
                    writeLength = writeLength - paddingBytes;
            }

            out.write(buffer, 0, writeLength);     // 向文件写入数据
        }

        out.close();

        return 0;
    }

        }
